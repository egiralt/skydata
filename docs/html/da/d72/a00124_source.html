355  private function BuildRenderDecorator ($renderOptions, &$methodInfo)
356  {
357  // Defaults
358  $methodInfo->showLoading = true;
359  // El siguiente patrón extraerá todos los campos de la línea de parámetros del método
360  if(preg_match_all(DOC_COMMENT_RENDER_TAG_PARAMETERS_PATTERN, $renderOptions, $paramMatches))
361  foreach ($paramMatches[0] as $order => $dumb)
362  {
363  $field = strtolower($paramMatches[1][$order]);
364  $value = $paramMatches[2][$order];
365  switch ($field)
366  {
367  case 'fullhtml' :
368  // Validaciones
369  if (!isset($methodInfo->contentType))
370  $methodInfo->contentType = SkyDataService::DATATYPE_HTML;
371  elseif ($methodInfo->contentType != SkyDataService::DATATYPE_HTML)
372  throw new \Exception('El método fue marcado con "fullHtml" pero el tipo de respuesta (contentType) no es HTML.', -100);
373  if (!empty($methodInfo->bind_model))
374  throw new \Exception('El método no puede estar asociado a un modelo de datos si ha sido marcado con "fullHtml" ', -100);
375  // Ya se puede confirmar que genera HTML
376  $methodInfo->tag->is_html = $value == 'true' ;
377  break;
378  case 'name' : $methodInfo->tag->name = $value; break;
379  case 'templateurl' : $methodInfo->tag->template_url = trim ($value,"'"); break;
380  case 'template' : $methodInfo->tag->template = trim ($value,"'"); break;
381  case 'presentation' : $methodInfo->tag->presentation = trim ($value,"'"); break;
382  case 'renderas' : $methodInfo->tag->render_as[] = $value; break;
383  case 'trigger' : $methodInfo->tag->trigger = $value; break;
384  case 'refresh' :
385  if (preg_match('/(\d*)(ms|s|m|h)/i', $value, $matches))
386  {
387  $count = 0;
388  switch ($matches[2])
389  {
390  case 'ms' : $count = $matches[1]; break; // milisegundos
391  case 's' : $count = $matches[1] * 1000; break; // segundos
392  case 'm' : $count = $matches[1] * 1000 * 60; break; // minutos
393  case 'h' : $count = $matches[1] * 1000 * 60 * 60; break;// horas
394  default:
395  throw new \Exception("El tiempo indicado para el refresh no es válido", -100);
396  }
397  $methodInfo->refresh = $count;
398  }
399  break;
400  case 'showloading': $methodInfo->showLoading = (empty($value) || $value == 'true'); break;
401  case 'prerenderview' : $methodInfo->preRenderView = trim ($value,"'"); break;
402  case 'onerrorview' : $methodInfo->onErrorView = trim ($value,"'"); break;
403  }
404  if (empty($methodInfo->tag->trigger))
405  $methodInfo->tag->trigger = 'init'; // Por defecto es "init"
406 
407  if (!isset($methodInfo->tag->name))
408  throw new \Exception("El servicio debe dar un nombre al atributo, elemento o clase (use name en el métod {$methodInfo->name})", 1);
409 
410  }
411  else
412  throw new \Exception("Error de sintaxis definiendo 'renderTag'.", -100);
413 
414  //echo "<pre>";print_r ($methodInfo); die();
415  }
