332  default:

340  // sin hacer nada mas.. Lo más veloz posible
341  $this->GetDeliveryChannel()->DeliverFromLocalUrlResource ($url);

344  {
345  $parts = explode ('/', $name);
346  $result = $this->HandlePageRequest($match, str_replace('#', '', $parts[0]));
